# Now its is time to perform my anomaly detection on the 'Blood_Pressure_mmHg' column

  time_decompose(Blood_Pressure_mmHg, method = "stl", frequency = 12, trend = 60) %>%
  anomalize(remainder, method = "iqr") %>%
  time_recompose()
